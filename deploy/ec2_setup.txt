associate instance with a static ip (elasticip)
point DNS to the ip
grant flexible (HTTP) access in the DNS

grant the instance an IAM role with ASM access
create postgres secret

sudo yum update -y
sudo yum install docker -y
sudo systemctl start docker
sudo systemctl enable docker
sudo usermod -aG docker ec2-user # Add user to docker group to avoid using sudo for docker commands

sudo curl -L "https://github.com/docker/compose/releases/latest/download/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose

sudo docker volume create pgdata

after deploying for the first time:
manually setup postgres user/pass (use secret in ASM) and database:
    CREATE DATABASE crossword_db;
    CREATE USER <user> WITH PASSWORD '<password>';
    ALTER DATABASE crossword_db OWNER TO <user>;

setting up an SSL certificate:
    docker volume create webroot
    docker volume create letsencrypt
    docker-compose down
    docker-compose up -d nginx_certbot_init (to run only certbot init)
    docker run --rm -v letsencrypt:/etc/letsencrypt -v webroot:/var/www/certbot certbot/certbot certonly --webroot -w /var/www/certbot -d conradscrossword.<com/dev> -d www.conradscrossword.<com/dev> --email conradkuklinsky@gmail.com --agree-tos --no-eff-email
    docker stop nginx_certbot_init
